## üìù Jenkinsfile - Pipeline Completo

### Pipeline Declarativo (Recomendado)

```groovy
// Jenkinsfile

pipeline {
    agent any

    // Vari√°veis de ambiente
    environment {
        APP_NAME = 'pedidos-api'
        DOCKER_REGISTRY = 'registry.empresa.com'
        IMAGE_NAME = "${DOCKER_REGISTRY}/${APP_NAME}"
        KUBECONFIG = credentials('kubeconfig')
        SONARQUBE_TOKEN = credentials('sonarqube-token')
    }

    // Ferramentas
    tools {
        maven 'Maven3'
        jdk 'JDK17'
    }

    // Op√ß√µes do pipeline
    options {
        buildDiscarder(logRotator(numToKeepStr: '10'))
        timestamps()
        timeout(time: 30, unit: 'MINUTES')
        disableConcurrentBuilds()
    }

    // Triggers (opcional)
    triggers {
        pollSCM('H/5 * * * *')  // Poll a cada 5 minutos
        // Ou usar webhook do GitHub
    }

    // Par√¢metros (opcional)
    parameters {
        choice(
            name: 'ENVIRONMENT',
            choices: ['dev', 'staging', 'prod'],
            description: 'Deploy environment'
        )
        booleanParam(
            name: 'SKIP_TESTS',
            defaultValue: false,
            description: 'Skip tests?'
        )
    }

    stages {
        // ========================================
        // Stage 1: Checkout
        // ========================================
        stage('Checkout') {
            steps {
                echo "üîÑ Checking out code from ${env.GIT_BRANCH}"

                checkout([
                    $class: 'GitSCM',
                    branches: [[name: '*/main']],
                    userRemoteConfigs: [[
                        url: 'https://github.com/empresa/pedidos-api.git',
                        credentialsId: 'github-token'
                    ]]
                ])

                script {
                    env.GIT_COMMIT_SHORT = sh(
                        script: "git rev-parse --short HEAD",
                        returnStdout: true
                    ).trim()
                }
            }
        }

        // ========================================
        // Stage 2: Build
        // ========================================
        stage('Build') {
            steps {
                echo "üî® Building application"

                sh '''
                    mvn clean package -DskipTests \
                        -Dmaven.test.skip=true \
                        -B -e
                '''
            }

            post {
                success {
                    archiveArtifacts artifacts: 'target/*.jar', fingerprint: true
                }
            }
        }

        // ========================================
        // Stage 3: Tests (Paralelo)
        // ========================================
        stage('Tests') {
            when {
                expression { params.SKIP_TESTS == false }
            }

            parallel {
                stage('Unit Tests') {
                    steps {
                        echo "üß™ Running unit tests"

                        sh 'mvn test -B'
                    }

                    post {
                        always {
                            junit 'target/surefire-reports/*.xml'

                            // JaCoCo coverage
                            jacoco(
                                execPattern: 'target/*.exec',
                                classPattern: 'target/classes',
                                sourcePattern: 'src/main/java',
                                exclusionPattern: 'src/test*'
                            )
                        }
                    }
                }

                stage('Integration Tests') {
                    steps {
                        echo "üß™ Running integration tests"

                        sh 'mvn verify -P integration-tests -DskipUnitTests=true -B'
                    }

                    post {
                        always {
                            junit 'target/failsafe-reports/*.xml'
                        }
                    }
                }
            }
        }

        // ========================================
        // Stage 4: Code Quality
        // ========================================
        stage('SonarQube Analysis') {
            steps {
                echo "üìä Running SonarQube analysis"

                withSonarQubeEnv('SonarQube') {
                    sh '''
                        mvn sonar:sonar \
                            -Dsonar.projectKey=${APP_NAME} \
                            -Dsonar.projectName=${APP_NAME} \
                            -Dsonar.host.url=${SONAR_HOST_URL} \
                            -Dsonar.token=${SONARQUBE_TOKEN}
                    '''
                }
            }
        }

        stage('Quality Gate') {
            steps {
                echo "üö¶ Waiting for quality gate"

                timeout(time: 5, unit: 'MINUTES') {
                    waitForQualityGate abortPipeline: true
                }
            }
        }

        // ========================================
        // Stage 5: Security Scan
        // ========================================
        stage('Security Scan') {
            steps {
                echo "üîí Running security scan"

                sh '''
                    # OWASP Dependency Check
                    mvn org.owasp:dependency-check-maven:check

                    # Trivy (vulnerabilities em dependencies)
                    trivy fs --exit-code 1 --severity HIGH,CRITICAL .
                '''
            }

            post {
                always {
                    publishHTML([
                        reportDir: 'target',
                        reportFiles: 'dependency-check-report.html',
                        reportName: 'OWASP Dependency Check'
                    ])
                }
            }
        }

        // ========================================
        // Stage 6: Docker Build
        // ========================================
        stage('Docker Build') {
            when {
                branch 'main'
            }

            steps {
                echo "üê≥ Building Docker image"

                script {
                    docker.withRegistry("https://${DOCKER_REGISTRY}", 'docker-credentials') {
                        def app = docker.build(
                            "${IMAGE_NAME}:${BUILD_NUMBER}",
                            "--build-arg JAR_FILE=target/*.jar ."
                        )

                        // Push com m√∫ltiplas tags
                        app.push("${BUILD_NUMBER}")
                        app.push("${GIT_COMMIT_SHORT}")
                        app.push("latest")
                    }
                }
            }
        }

        // ========================================
        // Stage 7: Deploy
        // ========================================
        stage('Deploy to Dev') {
            when {
                branch 'develop'
            }

            steps {
                echo "üöÄ Deploying to Dev environment"

                sh """
                    kubectl --kubeconfig=${KUBECONFIG} set image deployment/${APP_NAME} \
                        ${APP_NAME}=${IMAGE_NAME}:${BUILD_NUMBER} \
                        -n dev

                    kubectl --kubeconfig=${KUBECONFIG} rollout status deployment/${APP_NAME} -n dev --timeout=5m
                """
            }
        }

        stage('Deploy to Staging') {
            when {
                branch 'main'
            }

            steps {
                echo "üöÄ Deploying to Staging environment"

                sh """
                    kubectl --kubeconfig=${KUBECONFIG} set image deployment/${APP_NAME} \
                        ${APP_NAME}=${IMAGE_NAME}:${BUILD_NUMBER} \
                        -n staging

                    kubectl --kubeconfig=${KUBECONFIG} rollout status deployment/${APP_NAME} -n staging --timeout=5m
                """
            }
        }

        stage('Deploy to Production') {
            when {
                allOf {
                    branch 'main'
                    expression { params.ENVIRONMENT == 'prod' }
                }
            }

            steps {
                echo "‚è∏Ô∏è Waiting for approval"

                // Manual approval
                input(
                    message: 'Deploy to Production?',
                    ok: 'Deploy',
                    submitter: 'admin,tech-lead'
                )

                echo "üöÄ Deploying to Production environment"

                sh """
                    kubectl --kubeconfig=${KUBECONFIG} set image deployment/${APP_NAME} \
                        ${APP_NAME}=${IMAGE_NAME}:${BUILD_NUMBER} \
                        -n prod

                    kubectl --kubeconfig=${KUBECONFIG} rollout status deployment/${APP_NAME} -n prod --timeout=10m
                """
            }
        }

        // ========================================
        // Stage 8: Smoke Tests
        // ========================================
        stage('Smoke Tests') {
            when {
                branch 'main'
            }

            steps {
                echo "üß™ Running smoke tests"

                script {
                    sleep(time: 30, unit: 'SECONDS')  // Aguarda app ficar pronta

                    def response = sh(
                        script: """
                            curl -f -s -o /dev/null -w "%{http_code}" \
                                https://api-staging.empresa.com/actuator/health
                        """,
                        returnStdout: true
                    ).trim()

                    if (response != '200') {
                        error("Health check failed with status ${response}")
                    }

                    echo "‚úÖ Smoke tests passed"
                }
            }
        }
    }

    // ========================================
    // Post Actions
    // ========================================
    post {
        always {
            echo "üßπ Cleaning up workspace"
            cleanWs()
        }

        success {
            echo "‚úÖ Pipeline succeeded"

            slackSend(
                color: 'good',
                message: """
                    ‚úÖ Deploy Success
                    Job: ${env.JOB_NAME}
                    Build: #${env.BUILD_NUMBER}
                    Branch: ${env.GIT_BRANCH}
                    Commit: ${env.GIT_COMMIT_SHORT}
                    Duration: ${currentBuild.durationString}
                    URL: ${env.BUILD_URL}
                """,
                channel: '#deployments'
            )
        }

        failure {
            echo "‚ùå Pipeline failed"

            slackSend(
                color: 'danger',
                message: """
                    ‚ùå Deploy Failed
                    Job: ${env.JOB_NAME}
                    Build: #${env.BUILD_NUMBER}
                    Branch: ${env.GIT_BRANCH}
                    Error: ${currentBuild.result}
                    URL: ${env.BUILD_URL}
                """,
                channel: '#deployments'
            )

            emailext(
                subject: "Build Failed: ${env.JOB_NAME} #${env.BUILD_NUMBER}",
                body: """
                    Build failed.

                    Job: ${env.JOB_NAME}
                    Build: ${env.BUILD_NUMBER}
                    URL: ${env.BUILD_URL}
                """,
                to: 'team@empresa.com'
            )
        }

        unstable {
            echo "‚ö†Ô∏è Pipeline unstable"

            slackSend(
                color: 'warning',
                message: "‚ö†Ô∏è Build Unstable: ${env.JOB_NAME} #${env.BUILD_NUMBER}",
                channel: '#deployments'
            )
        }
    }
}
```
